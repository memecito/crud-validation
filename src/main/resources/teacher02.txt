-- Elimina la tabla si ya existe
DROP TABLE IF EXISTS teacher;

-- Crea la estructura de la tabla 'teacher'
CREATE TABLE teacher (
    id INT AUTO_INCREMENT PRIMARY KEY,
    person_id INT NOT NULL UNIQUE,
    branch VARCHAR(50),
    comments VARCHAR(255)
);
