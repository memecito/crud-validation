-- Elimina la tabla si ya existe para una ejecución limpia.
DROP TABLE IF EXISTS student_subject;

-- Crea la tabla de unión.
-- La clave primaria compuesta (student_id, subject_id) asegura que un estudiante
-- no pueda estar inscrito en la misma asignatura más de una vez.
CREATE TABLE student_subject (
    student_id INT NOT NULL,
    subject_id INT NOT NULL,
    PRIMARY KEY (student_id, subject_id)
);
